CREATE OR REPLACE PROCEDURE SP_FRAUD_DETECTION_ALL()
RETURNS STRING
LANGUAGE SQL
AS
$$
BEGIN

-- 1. SOC JUMP FRAUD
INSERT INTO CURATED.FRAUD_ALERTS
SELECT
    UUID_STRING(),
    'SOC_JUMP',
    session_id,
    user_id,
    charger_id,
    'SOC increased >30% within <2 minutes',
    CURRENT_TIMESTAMP()
FROM VALIDATED.SESSIONS_VAL
WHERE (end_soc - start_soc) > 30
AND DATEDIFF('minute', start_time, end_time) < 2;


-- 2. ENERGY vs CHARGER CAPACITY
INSERT INTO CURATED.FRAUD_ALERTS
SELECT
    UUID_STRING(),
    'ENERGY_MISMATCH',
    s.session_id,
    s.user_id,
    s.charger_id,
    'Energy exceeds charger capacity',
    CURRENT_TIMESTAMP()
FROM VALIDATED.SESSIONS_VAL s
JOIN VALIDATED.CHARGERS_FINAL c
ON s.charger_id = c.charger_id
WHERE s.energy_kwh > (c.max_kw * (s.duration_sec / 3600));


-- 3. GHOST CHARGING
INSERT INTO CURATED.FRAUD_ALERTS
SELECT
    UUID_STRING(),
    'GHOST_CHARGING',
    NULL,
    NULL,
    c.charger_id,
    'Charger active without session',
    CURRENT_TIMESTAMP()
FROM VALIDATED.CHARGERS_FINAL c
WHERE c.status = 'ACTIVE'
AND c.charger_id NOT IN (
    SELECT DISTINCT charger_id FROM VALIDATED.SESSIONS_VAL
);


-- 4. PARALLEL SESSIONS
INSERT INTO CURATED.FRAUD_ALERTS
SELECT
    UUID_STRING(),
    'PARALLEL_SESSION',
    s1.session_id,
    s1.user_id,
    s1.charger_id,
    'User running parallel sessions',
    CURRENT_TIMESTAMP()
FROM VALIDATED.SESSIONS_VAL s1
JOIN VALIDATED.SESSIONS_VAL s2
ON s1.user_id = s2.user_id
AND s1.session_id <> s2.session_id
AND s1.start_time < s2.end_time
AND s1.end_time > s2.start_time;


-- 5. IDLE TIME
INSERT INTO CURATED.FRAUD_ALERTS
SELECT
    UUID_STRING(),
    'IDLE_TIME',
    session_id,
    user_id,
    charger_id,
    'Excessive idle time after charging',
    CURRENT_TIMESTAMP()
FROM VALIDATED.SESSIONS_VAL
WHERE DATEDIFF('minute', end_time, CURRENT_TIMESTAMP()) > 60;


RETURN 'All Fraud Checks Completed Successfully';

END;
$$;

CALL SP_FRAUD_DETECTION_ALL();

CREATE OR REPLACE TABLE KPI_CHARGER_ANOMALY (

INSERT INTO CURATED.KPI_CHARGER_ANOMALY
SELECT
    CURRENT_TIMESTAMP() AS kpi_time,
    s.charger_id,

    COUNT(DISTINCT s.session_id) AS total_sessions,

    COUNT(DISTINCT f.session_id) AS flagged_sessions,

    (COUNT(DISTINCT f.session_id) / COUNT(DISTINCT s.session_id)) * 100 AS anomaly_score

FROM VALIDATED.SESSIONS_VAL s

LEFT JOIN CURATED.FRAUD_ALERTS f
    ON s.session_id = f.session_id

GROUP BY s.charger_id;

SELECT * 
FROM CURATED.KPI_CHARGER_ANOMALY
ORDER BY anomaly_score DESC;

CREATE OR REPLACE TABLE KPI_GRID_LOAD_INDEX (
    kpi_time TIMESTAMP,
    city STRING,
    zone STRING,
    total_stations NUMBER,
    balanced_stations NUMBER,
    grid_load_index FLOAT
);

INSERT INTO CURATED.KPI_GRID_LOAD_INDEX
WITH station_load AS (
    SELECT
        c.station_id,
        SUM(s.energy_kwh) / NULLIF(SUM(s.duration_sec)/3600,0) AS peak_load_kw
    FROM VALIDATED.SESSIONS_VAL s
    JOIN VALIDATED.CHARGERS_FINAL c
        ON s.charger_id = c.charger_id
    GROUP BY c.station_id
),

station_capacity AS (
    SELECT
        station_id,
        city,
        zone,
        capacity_kw,
        capacity_kw * 0.8 AS threshold_kw
    FROM VALIDATED.EV_STATIONS_VAL
),

final_calc AS (
    SELECT
        sc.city,
        sc.zone,
        sc.station_id,
        sc.capacity_kw,
        sc.threshold_kw,
        sl.peak_load_kw,

        CASE 
            WHEN sl.peak_load_kw <= sc.threshold_kw THEN 1
            ELSE 0
        END AS is_balanced
    FROM station_capacity sc
    LEFT JOIN station_load sl
        ON sc.station_id = sl.station_id
)

SELECT
    CURRENT_TIMESTAMP(),
    city,
    zone,
    COUNT(station_id) AS total_stations,
    SUM(is_balanced) AS balanced_stations,
    (SUM(is_balanced) / NULLIF(COUNT(station_id),0)) * 100 AS grid_load_index
FROM final_calc
GROUP BY city, zone;

SELECT * 
FROM CURATED.KPI_GRID_LOAD_INDEX
ORDER BY grid_load_index DESC;

    UPPER(TRIM(tariff_type)) AS tariff_type,

    COUNT(DISTINCT session_id) AS total_sessions,

    SUM(price) AS total_revenue,

    AVG(price) AS avg_revenue_per_session

FROM VALIDATED.SESSIONS_VAL

WHERE tariff_type IS NOT NULL

GROUP BY UPPER(TRIM(tariff_type));

SELECT *
FROM CURATED.KPI_ARPS
ORDER BY avg_revenue_per_session DESC;
